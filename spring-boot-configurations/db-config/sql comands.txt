DROP TABLE PRODUCT;

CREATE TABLE PRODUCT
(
    id       NUMBER GENERATED BY DEFAULT AS IDENTITY,
    name     VARCHAR2(255) NOT NULL,
    quantity NUMBER        NOT NULL,
    price    NUMBER(10, 2) NOT NULL,
    company  CHAR          NOT NULL,
    CONSTRAINT product_pk PRIMARY KEY (id)
);
